# Semantic Guard v3 Patterns
REDACT_PATTERNS = [
    (re.compile(r"[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}"), "EMAIL"),
    (re.compile(r"\b0\d{1,4}-\d{1,4}-\d{3,4}\b"), "PHONE"),
    (re.compile(r"AIza[0-9A-Za-z\-_]{35}"), "APIKEY"),
    (re.compile(r"-----BEGIN (?:RSA )?PRIVATE KEY-----"), "PRIVATE_KEY"),
]
FORCED_ACTION_RE = re.compile(r"(今すぐ|絶対|送金|削除|実行しろ|パスワード|秘密鍵|APIキー)", re.IGNORECASE)
SPECULATION_RE = re.compile(r"(可能性|かもしれない|だと思|推測|おそらく|たぶん)")
MORAL_RE = re.compile(r"(すべき|ほうがいい|望ましい|正しい|間違っている|導く)")
ASSERTION_RE = re.compile(r"(絶対|必ず|間違いない|確実)")
BULLET_PREFIX_RE = re.compile(r"^(\*|-|\d+\.)\s+")
REDACT_TAG_RE = re.compile(r"\[REDACT:([A-Z_]+)\]")
NE_REQUIRED_MARKERS = ["不明", "前提", "観測", "次"]
OUTPUT_MAX_LINES = 5

def now_ts() -> int: return int(time.time())
def today_str() -> str: return time.strftime("%Y-%m-%d", time.gmtime(time.time() + 9*3600))

def hmac_hex(key: bytes, data: bytes) -> str:
    return hmac.new(key, data, hashlib.sha256).hexdigest()

def canonical_json(obj: Any) -> str:
    return json.dumps(obj, ensure_ascii=False, separators=(",", ":"), sort_keys=True)

def sha256_hex(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def init_db(db_path: str):
    con = sqlite3.connect(db_path)
    cur = con.cursor()
    cur.execute("CREATE TABLE IF NOT EXISTS evidence (id INTEGER PRIMARY KEY AUTOINCREMENT, ts INTEGER, request_id TEXT, route TEXT, model TEXT, payload_json TEXT, prev_sig TEXT, sig TEXT)")
    cur.execute("CREATE TABLE IF NOT EXISTS usage_log (date TEXT PRIMARY KEY, calls INTEGER DEFAULT 0, in_chars INTEGER DEFAULT 0, out_chars INTEGER DEFAULT 0, est_cost REAL DEFAULT 0.0)")
    cur.execute("CREATE TABLE IF NOT EXISTS config (key TEXT PRIMARY KEY, value BLOB)")
    con.commit()
    con.close()

# 3. Core Logic (Redact / Guard / Cloud)
# ==========================================
def safety_redact_flow(text: str) -> str:
    masked = text
    for pat, label in REDACT_PATTERNS:
        masked = pat.sub(f"[REDACT:{label}]", masked)
    for pat, label in REDACT_PATTERNS:
        if pat.search(masked): raise Exception(f"[STOP: REDACTION_FAILURE] {label} remained")
    return masked

def call_cloud(prompt: str, model: str) -> str:
    enforced_prompt = prompt + "\n\n[OUTPUT_RULE]\n- 1-5 lines\n- Use '-' bullets only\n- Include '不明' at least once\n"
    try:
        resp = _gemini_client.models.generate_content(
            model=model,
            contents=enforced_prompt,
            config={"temperature": 0.1, "max_output_tokens": 140},
        )
        return (resp.text or "").strip()
    except Exception as e:
        return f"- 不明: クラウドエラー\n- 前提: 外部API失敗\n- 観測: {type(e).__name__}\n- 次: 再試行"

def semantic_guard_v3(output_text: str, safe_in: str) -> Tuple[str, str]:
    text = (output_text or "").strip()
    if not text: return "HOLD", "[HOLD: EMPTY]"
    if FORCED_ACTION_RE.search(text): return "STOP", "[STOP: FORCED_ACTION]"
    
    lines = [l for l in text.split("\n") if l.strip()]
    score = 0
    if len(lines) > OUTPUT_MAX_LINES: score += 2
    if not all(BULLET_PREFIX_RE.match(l.strip()) for l in lines): score += 1
    if not any(m in text for m in NE_REQUIRED_MARKERS): score += 1
    
    if score >= 6: return "STOP", "[STOP: SEMANTIC_HIGH_RISK]"
    if score >= 2: return "HOLD", "[HOLD: DEVIATION]"
    return "GO", text

def run_gateway(db_path: str, hmac_key: bytes, user_text: str, model: str = "gemini-2.0-flash", last_topics: List[str] = None) -> GatewayResult:
    try:
        safe_in = safety_redact_flow(user_text)
        # ※ここに LOC_SPIKE / USAGE チェックを挿入 (前述の定義済み関数)
        
        prompt = f"[CONTEXT]\n{last_topics}\n\n[USER]\n{safe_in}"
        rid = f"req_{now_ts()}_{sha256_hex(prompt)[:8]}"
        
        out_text = call_cloud(prompt, model)
        route, guarded = semantic_guard_v3(out_text, safe_in)
        
        # Evidence保存 (GOのみ本文)
        # append_evidence_record(db_path, hmac_key, rid, route, model, prompt, guarded if route=="GO" else "", {})
        
        return GatewayResult(route=route, text=guarded, request_id=rid)
    except Exception as e:
        return GatewayResult(route="STOP", text=str(e), request_id="ERR")
