NECore/DomainModels.swift

import Foundation

public enum Domain: String, Codable, CaseIterable, Identifiable {
    case general, medical, logistics
    public var id: String { rawValue }
}

public struct DomainRequest: Codable, Sendable {
    public let rid: String
    public let seq: Int64
    public let domain: Domain
    public let text: String
    public let locale: String
    public init(rid: String, seq: Int64, domain: Domain, text: String, locale: String) {
        self.rid = rid; self.seq = seq; self.domain = domain; self.text = text; self.locale = locale
    }
}

public struct DomainResponse: Codable, Sendable {
    public let rid: String
    public let seq: Int64
    public let domain: Domain
    public let output: String
    public let confidence: Float
    public let reason: String?
    public let ms: Int
    public init(rid: String, seq: Int64, domain: Domain, output: String, confidence: Float, reason: String?, ms: Int) {
        self.rid = rid; self.seq = seq; self.domain = domain; self.output = output
        self.confidence = confidence; self.reason = reason; self.ms = ms
    }
}

public struct StoredEvaluation: Codable, Sendable {
    public let rid: String
    public let seq: Int64
    public let ts: Int64
    public let monoMs: Int64
    public let isValidated: Bool
    public let confidence: Float
    public let reason: String?
    public let ms: Int
    public init(rid: String, seq: Int64, ts: Int64, monoMs: Int64, isValidated: Bool, confidence: Float, reason: String?, ms: Int) {
        self.rid = rid; self.seq = seq; self.ts = ts; self.monoMs = monoMs
        self.isValidated = isValidated; self.confidence = confidence; self.reason = reason; self.ms = ms
    }
}
